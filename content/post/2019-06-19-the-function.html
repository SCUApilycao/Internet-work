##想看看距离自己出生已经过去了多少秒

t<-Sys.time()
unclass(t)
## [1] 1560910841
g<-as.Date("2000-6-12")
difftime(t,g,units="secs")
## Time difference of 600142841 secs

##获取一天前的日期

t<-Sys.Date()
as.POSIXlt(t-1)
## [1] "2019-06-18 UTC"

##自定义时间格式

t<-Sys.Date()
format(t,format="%Y-%m-%d")
## [1] "2019-06-19"
format(t,format="%B-%d-%Y")
## [1] "六月-19-2019"
